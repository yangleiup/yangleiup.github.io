# call,apply用法及区别

call和apply都是用来改变this指向，第一次参数为函数运行时想要指定的this值

区别在于call接收第二个及以后参数，apply只接收俩参数，第二个参数可以是数组也可以是类数组，其实也可以是对象，后续的参数忽略不计。

严格模式下，函数的this值就是call和apply的第一个参数，非严格模式下，第一个参数值被指定为 null 或 undefined时this值会自动替换为指向全局对象，原始值则会被自动包装，也就是new Object()。

看两个demo

// 例子1：浏览器环境 非严格模式下
var doSth = function(a, b){
  console.log(this);
  console.log([a, b]);
}
doSth.apply(null, [1, 2]); // this是window  // [1, 2]
doSth.apply(0, [1, 2]); // this 是 Number(0) // [1, 2]
doSth.apply(true); // this 是 Boolean(true) // [undefined, undefined]
doSth.call(undefined, 1, 2); // this 是 window // [1, 2]
doSth.call('0', 1, {a: 1}); // this 是 String('0') // [1, {a: 1}]

Function.prototype.mycall = function(context) {
    context.fn = this;

# 模拟实现第三步

模拟代码已经完成 80%，还有两个小点要注意：

  1. this 参数可以传 null，当为 null 的时候，视为指向 window
  2. 函数是可以有返回值的！
// 第三版
Function.prototype.mycall = function (context) {
    var context = context || window;
    context.fn = this;

    var args = [];
    for(var i = 1, len = arguments.length; i < len; i++) {
        args.push('arguments[' + i + ']');
    }

    var result = eval('context.fn(' + args +')');

    delete context.fn
    return result;
}

// 测试一下
var value = 2;

var obj = {
    value: 1
}

function bar(name, age) {
    console.log(this.value);
    return {
        value: this.value,
        name: name,
        age: age
    }
}

bar.call(null); // 2

console.log(bar.mycall(obj, 'kevin', 18));
// 1

//    age: 18

截止第三版还有点小问题

  1. 没有参数类型等校验
  2. 可能绑定的对象中已经有fn了，保证fn名字的唯一性，或者若不唯一，怎么处理
  3. 执行函数用eval不友好

// 浏览器环境 非严格模式
function getGlobalObject(){
  return this;
}

function generateFunctionCode(argsArrayLength){
    var code = 'return arguments[0][arguments[1]](';
    for(var i = 0; i < argsArrayLength; i++){
        if(i > 0){
            code += ',';
        }
        code += 'arguments[2][' + i + ']';
    }
    code += ')';
    // return arguments[0][arguments[1]](arg1, arg2, arg3...)
    return code;
}

Function.prototype.mycall = function (context) {
  // 解决问题一：参数方法校验
  if(typeof this !== 'function'){
    throw new TypeError(this + ' is not a function');
  }

  if(typeof context === 'undefined' || context === null){
    context = getGlobalObject();
  }

  // 解决问题二：fn唯一 1.采用ES6 Sybmol() 独一无二的 2.Math.random()模拟实现独一无二的key 3.直接用生成时间戳
  var __fn = '__' + new Date().getTime();

  // 万一还是有 先存储一份，删除后，再恢复该值
  var originalVal = context[__fn];
  // 是否有原始值
  var hasOriginalVal = context.hasOwnProperty(__fn);
  // this表示调用call的函数
  context[__fn] = this;

  
  var args = [];
  for(var i = 0, len = arguments.length; i < len; i++) {
    args[i] = arguments[i];
  }

// 解决问题三：采用new Function()来生成执行函数
  var code = generateFunctionCode(args.length);
  var result = (new Function(code))(context, __fn, args);
  //  ES6版
  // var result = context[__fn](...args);

  delete context.fn
  if(hasOriginalVal){
    context[__fn] = originalVal;
  }
  return result;
}
